部署到 github page #

在 git 中新增如下钩子即可

#!/bin/bash

# 确保脚本抛出遇到的错误
set -e

# 生成静态文件
file="docs/.vitepress/dist"

# 将无用文件先删除
if [ -f $file ] ; then
    rm -rf $file
fi
yarn docs:build

# 进入生成的文件夹
cd $file
git init
git add -A
git commit -m 'deploy'
# 如果你想要部署到 https://<USERNAME>.github.io
git push -f git@github.com:threedayAAAAA/threedayAAAAA.github.io.git master:main

# 本地部署
cd ..
cp -r dist/. d:/nginx/html
cd d:/nginx
start nginx

exit 0
